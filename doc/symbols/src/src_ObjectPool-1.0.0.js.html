 10 /** 
 11  * @namespace Namespace of alumican libs.
 12  */
 13 var ALUMICAN;
 14 if (!ALUMICAN) ALUMICAN = {}; 
 15 
 16 /**
 17  * ObjectPool class.
 18  * @class
 19  * @param {function} onRequireItem This callback function is called when new item is required.
 20  * @param {function} onDestroyItem This callback function is called when item is disposed.
 21  * @param {integer} initCount Init item count.
 22  * @param {integer} growthCount Growth item count.
 23  *
 24  * @property {integer} initCount Init item count.
 25  * @property {integer} growthCount Growth item count.
 26  * @property {function} onRequireItem This callback function is called when new item is required.
 27  * @property {function} onDestroyItem This callback function is called when item is disposed.
 28  * @property {array} items Array of items.
 29  * @property {integer} index Current index of items.
 30  *
 31  * @return void
 32  */
 33 ALUMICAN.ObjectPool = function(onRequireItem, onDestroyItem, initCount, growthCount)
 34 {
 35 	this.onRequireItem;
 36 	this.onDestroyItem;
 37 	this.initCount;
 38 	this.growthCount;
 39 	this.items;
 40 	this.index;
 41 
 42 	//----------------------------------------
 43 	//call construntor
 44 	this._initialize.apply(this, arguments);
 45 }
 46 
 47 /**
 48  * Version information.
 49  * @static
 50  * @constant
 51  */
 52 ALUMICAN.ObjectPool.version = "1.0.0";
 53 
 54 ALUMICAN.ObjectPool.prototype =
 55 {
 56 	/**
 57 	 * Get item from object pool.
 58 	 * @return {object} Item
 59 	 */
 60 	getItem : function()
 61 	{
 62 		if (this.index > 0) return this.items[--this.index];
 63 		for (var i = 0; i < this.growthCount; ++i)
 64 		{
 65 			this.items.unshift(this.onRequireItem());
 66 		}
 67 		this.index = this.growthCount;
 68 		return this.getItem();
 69 	},
 70 	
 71 	/**
 72 	 * Return item to object pool.
 73 	 * @param {object} Item.
 74 	 * @return void
 75 	 */
 76 	returnItem : function(item)
 77 	{
 78 		this.items[this.index++] = item;
 79 	},
 80 	
 81 	/**
 82 	 * Optimize object pool size.
 83 	 * @return void
 84 	 */
 85 	reduce : function()
 86 	{
 87 		var n = this.index;
 88 		for (var i = 0; i < n; ++i)
 89 		{
 90 			this.onDestroyItem(this.items.pop());
 91 		}
 92 		this.index = 0;
 93 	},
 94 	
 95 	/**
 96 	 * Destroy object pool.
 97 	 * @return void
 98 	 */
 99 	destroy : function()
100 	{
101 		var n = this.items.length;
102 		for (var i = 0; i < n; ++i)
103 		{
104 			this.onDestroyItem(this.items[i]);
105 		}
106 		this.items = null;
107 		this.onRequireItem = null;
108 		this.onDestroyItem = null;
109 	},
110 	
111 	//----------------------------------------
112 	/**
113 	 * #@+
114 	 * @private
115 	 */
116 	_initialize : function(onRequireItem, onDestroyItem, initCount, growthCount)
117 	{
118 		this.onRequireItem = (onRequireItem != undefined) ? onRequireItem : function() { return {} };
119 		this.onDestroyItem = (onDestroyItem != undefined) ? onDestroyItem : function(item) {};
120 		this.initCount     = (initCount     != undefined) ? initCount     : 100;
121 		this.growthCount   = (growthCount   != undefined) ? growthCount   : 50;
122 		this.items = new Array(this.initCount);
123 		for (var i = 0; i < this.initCount; ++i)
124 		{
125 			this.items[i] = this.onRequireItem();
126 		}
127 		this.index = this.initCount;
128 	}
129 	
130 	/** #@- */
131 }
